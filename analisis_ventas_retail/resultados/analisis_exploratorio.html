# Anlisis por categora
ventas_categoria = df.groupby('Category').agg({
    'Sales': 'sum',
    'Profit': 'sum',
    'Order ID': 'nunique'
}).round(2)

ventas_categoria.columns = ['Ventas', 'Ganancia', 'Pedidos']
ventas_categoria = ventas_categoria.sort_values('Ventas', ascending=False)

print("\n VENTAS POR CATEGORA:")
print(ventas_categoria)

# Visualizacin
plt.figure(figsize=(10, 6))
ventas_categoria['Ventas'].plot(kind='bar', color=['#2ecc71', '#3498db', '#e74c3c'])
plt.title('Ventas Totales por Categora', fontsize=16, fontweight='bold')
plt.xlabel('Categora', fontsize=12)
plt.ylabel('Ventas ($)', fontsize=12)
plt.xticks(rotation=0)
plt.grid(axis='y', alpha=0.3)

# Guardar grfico
plt.tight_layout()
plt.savefig('../resultados/ventas_por_categoria.png', dpi=300, bbox_inches='tight')
plt.show()

# Ventas mensuales
ventas_mensuales = df.groupby(['Year', 'Month'])['Sales'].sum().reset_index()

plt.figure(figsize=(14, 6))
for year in ventas_mensuales['Year'].unique():
    data_year = ventas_mensuales[ventas_mensuales['Year'] == year]
    plt.plot(data_year['Month'], data_year['Sales'], marker='o', label=f'Ao {year}', linewidth=2)

plt.title('Tendencia de Ventas Mensuales', fontsize=16, fontweight='bold')
plt.xlabel('Mes', fontsize=12)
plt.ylabel('Ventas ($)', fontsize=12)
